program sort;
let i, j, x, min, a[10]: int;

main() {
	a[0] = 6;
	a[1] = 4;
	a[2] = 9;
	a[3] = 10;
	a[4] = 1;
	a[5] = 3;
	a[6] = 2;
	a[7] = 8;
	a[8] = 5;
	a[9] = 7;

	x = 0;
	for i = 0 to 10 {
		x = i;
		min = a[i];
		for j = i+1 to 10 {
			if(a[j] < min){
				min = a[j];
				x = j;
			}
		}
		a[x] = a[i];
		a[i] = min;
	}
}
